    /**
     *
     * @param growth
     * @param interfere
     * @param time
     * @return
     */
   public static int getPlantLushSums(int[][] growth, int[] interfere, int time) {
        int loop = 1;
        int row = growth.length;
        int col = growth[0].length;
        int[][] array = new int[row][];
        for (int i = 0; i < array.length; i++) {
            array[i] = growth[i].clone();
        }

        while (loop <= time) {
            int[][] array1 = getRole1Array(array);
            int[][] array2 = getRole2Array(array);

            for (int i = 0; i < row; i++) {
                for (int j = 0; j < col; j++) {
                    array[i][j] += (array1[i][j] + array2[i][j]);
                }
            }

            if (interfere[0] == loop) {
                array[interfere[1]][interfere[2]] += interfere[3];
            }

            loop++;
        }

        int sum = 0;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                sum += array[i][j];
            }
        }

        return sum;
    }

    /**
     * 演变规则一：growth[i][j]上下左右的元素和大于等于21时，growth[i][j]的值减5；
     *
     * @param growth
     * @return
     */
    private static int[][] getRole1Array(int[][] growth) {
        int row = growth.length;
        int col = growth[0].length;
        int[][] array1 = new int[row][col];
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                int sum = 0;
                if (i - 1 >= 0) {
                    sum += growth[i - 1][j];
                }
                if (i + 1 < row) {
                    sum += growth[i + 1][j];
                }
                if (j - 1 >= 0) {
                    sum += growth[i][j - 1];
                }
                if (j + 1 < col) {
                    sum += growth[i][j + 1];
                }

                if (sum >= 21) {
                    array1[i][j] = -5;
                }
            }
        }

        return array1;
    }

    /**
     * 演变规则二：growth[i][j]大于0时，上下左右的元素各加1；
     *
     * @param growth
     * @return
     */
    private static int[][] getRole2Array(int[][] growth) {
        int row = growth.length;
        int col = growth[0].length;
        int[][] array2 = new int[row][col];
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (i - 1 >= 0 && growth[i][j] > 0) {
                    array2[i - 1][j] += 1;
                }
                if (i + 1 < row && growth[i][j] > 0) {
                    array2[i + 1][j] += 1;
                }
                if (j - 1 >= 0 && growth[i][j] > 0) {
                    array2[i][j - 1] += 1;
                }
                if (j + 1 < col && growth[i][j] > 0) {
                    array2[i][j + 1] += 1;
                }
            }
        }

        return array2;
    }
